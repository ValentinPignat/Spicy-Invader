276 private List<Highscore> GetHighscore()
277 {
278 List <Highscore> highscores = new List<Highscore>();
279 string _content = "";
280 int score = 0;
281
282 // If file doesn't exist ...
283 while (!File.Exists(HIGHSCORE_PATH))
284 {
285 // ... try to create it
286 try
287 {
288 // https://stackoverflow.com/questions/5156254/closing-a-file-after-file-create
289 // https://stackoverflow.com/questions/66537978/c-sharp-system-io-ioexception-file-cannot-be-accessed-because-it-is-accessed-by
290 // Create and CLOSE file to avoid used by other process error when reading content
291 FileStream file = File.Create(HIGHSCORE_PATH);
292 file.Close();
293 }
294 // If the path is incorect, warning and exit with empty highscore
295 catch
296 {
297 return new List<Highscore>();
298 }
299 }
300
301 // Get txt file content as a string and then split it into lines
302 _content = File.ReadAllText(HIGHSCORE_PATH);
303 List<string> lines = _content.Split('\n').ToList();
304
305 // For each of the lines add the score to the _highscores
306 foreach (string line in lines)
307 {
308 // Split username and score
309 string[] splited = line.Split(SEPARATOR);
310
311 // Test that two elements are on the line
312 if (splited.Length == 2) {
313
314 // Test if highscore has a valid value
315 if (Int32.TryParse(splited[INDEX_SCORE], out score))
316 {
317 Highscore highscore = new Highscore(score: score, username: splited[INDEX_USERNAME]);
318 highscores.Add(highscore);
319 }
320 }
321
322 }
323 // If file found or created return highscores
324 return highscores;
325 }

585 public void GameOver(int score) {
586
587 Console.Clear();
588
589 // Player name
590 string username = "";
591
592 // Display title
593 WriteCenterHorizontal(toDisplay: GAME_OVER_TITLE, top: MARGIN_TOP * 1);
594
595 // Delay before taking input to not take involuntary input
596 Thread.Sleep(1000);
597
598 // Flush keyboard buffer
599 //https://stackoverflow.com/questions/978091/c-sharp-or-net-flushing-keyboard-buffer
600 while (Console.KeyAvailable) { Console.ReadKey(intercept: true); }
601
602 // Display prompt for input
603 WriteCenterHorizontal(toDisplay: GAME_OVER_PROMPT, top: MARGIN_TOP * 2 + GAME_OVER_TITLE.Split('\n').Length);
604
605 // Read input and add to highscores
606 // Username is cut if longer that MAX_NAME_SIZE
607 username = Console.ReadLine();
608 username = username.Trim();
609 if (username.Length > MAX_NAME_SIZE)
610 {
611 username = username.Substring(0, MAX_NAME_SIZE);
612 }
613 AddToHighscore(score: score, username: username);
