335 private void GameStartLoop() {
336
337 // Loop : Check for pause > Player action > Missile upate > Enemy update > Colisions check > Dispose of dead objects
338 while (_gameRunning)
339 {
340 // If V is pressed enter pause
341 if (Keyboard.IsKeyDown(Key.V))
342 {
343 _soundManager.MenuSound();
344 _pauseMenu.Pause();
345 DrawAll();
346 }
347
348 // Player update every PLAYER_SPEED cycles
349 if (_playerCycle == PLAYER_SPEED)
350 {
351 PlayerControl();
352 _playerCycle = 0;
353 }
354 else
355 {
356 _playerCycle++;
357 }
358
359 // Missiles update and MOVE every MISSILE_SPEED cycles
360 if (_missileCycle == MISSILES_SPEED)
361 {
362 _player.UpdateMissile(moving: true);
363 _enemyBlock.UpdateMissile(moving: true);
364 _missileCycle = 0;
365 }
366 // Missile update without moving
367 else
368 {
369 _missileCycle++;
370 _player.UpdateMissile(moving: false);
371 _enemyBlock.UpdateMissile(moving: false);
372 }
373
374 // CheckColision() for player's missile(s);
375 foreach (Missile missile in _player._missilesList)
376 {
377 CheckColision(missile: missile, target: _collisionObjects, targetMissile: _enemyBlock._missilesList);
378 }
379
380 // CheckColision() for enemy missile(s);
381 foreach (Missile missile in _enemyBlock._missilesList)
382 {
383 CheckColision(missile: missile, target: _collisionObjects);
384 }
385
386 // Check for "dead" objects (hp = 0)
387 DeadUpdate(_collisionObjects);
388
389 // If _enemyBlock is empty end game
390 if (_enemyBlock.IsEmpty())
391 {
392 _gameRunning = false;
393 }
394
395 // Display score
396 DisplayScoreHp(score: _score, hp: _player.Hp);
397
398 // Enemy speed goes up when they go down, ratio can be changed
399 if (_enemyCycle >= (double)ENEMY_SPEED - ((_enemyBlock.LinesDown > 1) ? (double)_enemyBlock.LinesDown / RATIO_LINEDOWN_ACCELETATION : 1))
400 {
401 _enemyBlock.Update();
402 _enemyCycle = 0;
403 }
404 else
405 {
406 _enemyCycle += 1;
407 }
408
409 // Time between each cycle
410 Thread.Sleep(CYCLE_SPEED);
411
412 // If game is not running bc enemyBlock is empty and easy mode is off
413 if (_gameRunning == false && !_easymode && _player.Hp > 0)
414 {
415
416 // Respawn ennemies and continue
417 SpawnEnemies();
418 _gameRunning = true;
419 }
420 } // End of Game loop
421
422 // After game end add score to highscore
423 _pauseMenu.GameOver(score: _score);
424
425 }

462 private void CheckColision(Missile missile, List<GameObject> target, List<Missile> targetMissile = null)
463 {
464
465 // Foreach GameObject ...
466 foreach (GameObject gameObj in target)
467 {
468 if (gameObj != null)
469 {
470
471 // ... if missile is on any coordinate in his square hitbox ...
472 if (missile.X >= gameObj.X && missile.X <= gameObj.X + gameObj.Width && missile.Y >= gameObj.Y && missile.Y <= gameObj.Y + gameObj.Height)
473 {
474 if (missile.CollisionStatus != gameObj.CollisionStatus)
475 {
476 _soundManager.CollisionSound();
477 // ... deal one damage to both objects
478 if (missile.Hp > 0) { missile.Hp--; }
479 if (gameObj.Hp > 0) { gameObj.Hp--; }
480 }
481 return;
482 }
483 }
484 }
485
486 if (targetMissile != null) {
487 // Foreach Missile ...
488 foreach (Missile tarMissile in targetMissile)
489 {
490 if (tarMissile != null)
491 {
492
493 // ... if missile is on any coordinate in missile target square hitbox ...
494 if (missile.X >= tarMissile.X && missile.X <= tarMissile.X + tarMissile.Width && missile.Y >= tarMissile.Y && missile.Y <= tarMissile.Y + tarMissile.Height)
495 {
496 if (missile.CollisionStatus != tarMissile.CollisionStatus)
497 {
498 _soundManager.CollisionSound();
499 // ... deal one damage to both objects
500 if (missile.Hp > 0) { missile.Hp--; }

541 {
