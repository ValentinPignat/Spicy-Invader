263 private List<Highscore> GetHighscore()
264 {
265 List <Highscore> highscores = new List<Highscore>();
266 string _content = "";
267 int score = 0;
268
269 // If file doesn't exist ...
270 while (!File.Exists(HIGHSCORE_PATH))
271 {
272 // ... try to create it
273 try
274 {
275 // https://stackoverflow.com/questions/5156254/closing-a-file-after-file-create
276 // https://stackoverflow.com/questions/66537978/c-sharp-system-io-ioexception-file-cannot-be-accessed-because-it-is-accessed-by
277 // Create and CLOSE file to avoid used by other process error when reading content
278 FileStream file = File.Create(HIGHSCORE_PATH);
279 file.Close();
280 }
281 // If the path is incorect, warning and exit with empty highscore
282 catch
283 {
284 return new List<Highscore>();
285 }
286 }
287
288 // Get txt file content as a string and then split it into lines
289 _content = File.ReadAllText(HIGHSCORE_PATH);
290 List<string> lines = _content.Split('\n').ToList();
291
292 // For each of the lines add the score to the _highscores
293 foreach (string line in lines)
294 {
295 // Split username and score
296 string[] splited = line.Split(SEPARATOR);
297
298 // Test that two elements are on the line
299 if (splited.Length == 2) {
300
301 // Test if highscore has a valid value
302 if (Int32.TryParse(splited[INDEX_SCORE], out score))
303 {
304 Highscore highscore = new Highscore(score: score, username: splited[INDEX_USERNAME]);
305 highscores.Add(highscore);
306 }
307 }
308
309 }
310 // If file found or created return highscores
311 return highscores;
312 }
